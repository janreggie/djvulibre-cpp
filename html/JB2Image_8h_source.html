728 JB2Dict::JB2Codec::get_direct_context( unsigned char const * const up2,
729  unsigned char const * const up1,
730  unsigned char const * const up0,
731  const int column)

734  (up2[column ] << 8) |
735  (up2[column + 1] << 7) |
736  (up1[column - 2] << 6) |
737  (up1[column - 1] << 5) |
738  (up1[column ] << 4) |
739  (up1[column + 1] << 3) |
