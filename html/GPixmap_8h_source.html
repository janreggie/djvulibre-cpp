442 inline int
443 operator==(const GPixel & p1, const GPixel & p2)

445  return p1.r==p2.r && p1.g==p2.g && p1.b==p2.b;

454 inline unsigned int
455 hash(const GPixel &p)
456 {
457  unsigned int x = (p.b<<16)|(p.g<<8)|(p.r);
