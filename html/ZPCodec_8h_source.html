362 ZPCodec::encoder(int bit, BitContext &ctx)
363 {
364  unsigned int z = a + p[ctx];
365  if (bit != (ctx & 1))
366  {
367  encode_lps(ctx, z);
368  }else if (z >= 0x8000)
369  {
370  encode_mps(ctx, z);
371  }else
372  {
373  a = z;
374  }
375 }
