370 // The ERR_MSG(x) macro is intended to permit automated checking of the
371 // externalized error message names against the source code. It has no
372 // effect on the executed program. It should be used to surround each
373 // message name that will need to be looked up in the external message
374 // files. In particular, it should use on all strings passed to G_THROW.
375 #ifndef HAS_CTRL_C_IN_ERR_MSG
376 # define HAS_CTRL_C_IN_ERR_MSG 1
377 #endif
378 #ifndef ERR_MSG
379 # if HAS_CTRL_C_IN_ERR_MSG
380 // This hack allows for the coexistence of internationalized
381 // and non-internationalized code. All internationalized error
382 // message names are prefixed with a ctrl-c. Only these will
383 // be looked for in the message files. Messages that do no
384 // start with a ctrl-c will remain untranslated.
385 # define ERR_MSG(x) "\003" x
386 # else
387 # define ERR_MSG(x) x
388 # endif
389 #endif
