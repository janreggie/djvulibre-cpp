636 inline void
637 GBitmap::append_run(unsigned char *&data, int count)
638 {
639  if (count < RUNOVERFLOWVALUE)
640  {
641  data[0] = count;
642  data += 1;
643  }
644  else if (count <= MAXRUNSIZE)
645  {
646  data[0] = (count>>8) + GBitmap::RUNOVERFLOWVALUE;
647  data[1] = (count & 0xff);
648  data += 2;
649  }
650  else
651  {
652  append_long_run(data, count);
653  }
654 }
